## 二叉树转换成中序链表

````js
    1
  2   3
4   5

function transformTreeToLinkedNode(root){
  // 4 2 5 1 3
  if(root === null)
    return null
  if(root.left === null && root.right === null)
    return root

  let nodeL,nodeR

  nodeL = transformTreeToLinkedNode(root.left)
  
  nodeR = transformTreeToLinkedNode(root.right)  

  if(nodeL != null)
    nodeL.left = root
  else
    nodeL = root
  
  root.left = nodeR

  return nodeL
}
````
